16  package org.onehippo.forge.camel.converter.json;
17  
18  import java.io.InputStream;
19  import java.nio.charset.StandardCharsets;
20  import java.util.List;
21  
22  import org.apache.camel.builder.RouteBuilder;
23  import org.apache.camel.test.junit5.CamelTestSupport;
24  import org.easymock.EasyMock;
25  import org.json.JSONObject;
26  import org.junit.jupiter.api.Test;
27  import org.onehippo.repository.scheduling.RepositoryJobExecutionContext;
28  
29  import static org.junit.jupiter.api.Assertions.assertEquals;
30  import static org.junit.jupiter.api.Assertions.assertNotNull;
31  import static org.junit.jupiter.api.Assertions.assertTrue;
32  
33  /**
34   * Integration test that exercises type converter lambdas in the generated
35   * JSONConverterLoader and RepositoryJobExecutionContextConverterLoader
36   * via Camel's type conversion infrastructure.
37   */
38  class CamelTypeConverterIntegrationTest extends CamelTestSupport {
39  
40      @Test
41      void convertJsonObjectToString() {
42          JSONObject json = new JSONObject();
43          json.put("key", "value");
44  
45          String result = context.getTypeConverter().convertTo(String.class, json);
46  
47          assertNotNull(result);
48          assertTrue(result.contains("key"));
49          assertTrue(result.contains("value"));
50      }
51  
52      @Test
53      void convertJsonObjectToInputStream() throws Exception {
54          JSONObject json = new JSONObject();
55          json.put("key", "value");
56  
57          InputStream result = context.getTypeConverter().convertTo(InputStream.class, json);
58  
59          assertNotNull(result);
60          String content = new String(result.readAllBytes(), StandardCharsets.UTF_8);
61          assertTrue(content.contains("key"));
62          assertTrue(content.contains("value"));
63      }
64  
65      @Test
66      void convertStringToJsonObject() {
67          String jsonStr = "{\"key\":\"value\",\"num\":42}";
68  
69          JSONObject result = context.getTypeConverter().convertTo(JSONObject.class, jsonStr);
70  
71          assertNotNull(result);
72          assertEquals("value", result.getString("key"));
73          assertEquals(42, result.getInt("num"));
74      }
75  
76      @Test
77      void convertRepositoryJobExecutionContextToJsonObject() {
78          RepositoryJobExecutionContext ctx = EasyMock.createMock(RepositoryJobExecutionContext.class);
79          EasyMock.expect(ctx.getAttributeNames()).andReturn(List.of("attr1", "attr2")).anyTimes();
80          EasyMock.expect(ctx.getAttribute("attr1")).andReturn("val1").anyTimes();
81          EasyMock.expect(ctx.getAttribute("attr2")).andReturn("val2").anyTimes();
82          EasyMock.replay(ctx);
83  
84          JSONObject result = context.getTypeConverter().convertTo(JSONObject.class, ctx);
85  
86          assertNotNull(result);
87          assertEquals("val1", result.getString("attr1"));
88          assertEquals("val2", result.getString("attr2"));
89          EasyMock.verify(ctx);
90      }
91  
92      @Override
93      protected RouteBuilder createRouteBuilder() {
94          return new RouteBuilder() {
95              @Override
96              public void configure() {
97              }
98          };
99      }
100 }
