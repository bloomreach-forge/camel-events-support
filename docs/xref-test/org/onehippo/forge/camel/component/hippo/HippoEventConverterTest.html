39      @Test
40      public void testHippoEvent() throws Exception {
41          final long timestamp = System.currentTimeMillis();
42  
43          HippoEvent event = new HippoEvent("application1");
44          event.action("action1");
45          event.category("category1");
46          event.message("message1");
47          event.result("result1");
48          event.system(true);
49          event.timestamp(timestamp);
50          event.user("user1");
51          event.set("attr1", "value1");

58          assertEquals("action1", json.get("action"));
59          assertEquals("category1", json.get("category"));
60          assertEquals("message1", json.get("message"));
61          assertEquals("result1", json.get("result"));
62          assertEquals(true, json.get("system"));
63          assertEquals(timestamp, json.get("timestamp"));
64          assertEquals("user1", json.get("user"));
65          assertEquals("value1", json.get("attr1"));
66          assertEquals("value2", json.get("attr2"));
67          assertEquals(true, json.get("sealed"));
68      }

103     @Test
104     public void testHippoWorkflowEvent() throws Exception {
105         final long timestamp = System.currentTimeMillis();
106         final List<String> arguments = Arrays.asList("arg1", "arg2");
107         final Exception exception = new Exception();
108 
109         HippoWorkflowEvent event = new HippoWorkflowEvent();
110         event.action("action1");
111         event.message("message1");
112         event.result("result1");
113         event.system(true);
114         event.timestamp(timestamp);
115         event.user("user1");
116         event.set("attr1", "value1");
117         event.set("attr2", "value2");
118         event.arguments(arguments);
119         event.className("example.Workflow");
120         event.exception(exception);
121         event.interaction("interaction1");
122         event.interactionId("interactionId1");
123         event.returnType("returnType1");
124         event.returnValue("returnValue1");
125         event.subjectId("subjectId1");
126         event.subjectPath("subjectPath1");
127         event.workflowCategory("workflowCategory1");
128         event.workflowName("workflowName1");
129         event.sealEvent();
130 
131         JSONObject json = HippoEventConverter.toJSONObject(event);
132         assertEquals(HippoWorkflowEvent.class.getName(), json.get("_eventClassName"));
133         assertEquals("repository", json.get("application"));
134         assertEquals("action1", json.get("action"));
135         assertEquals("workflow", json.get("category"));
136         assertEquals("message1", json.get("message"));
137         assertEquals("result1", json.get("result"));
138         assertEquals(true, json.get("system"));
139         assertEquals(timestamp, json.get("timestamp"));
140         assertEquals("user1", json.get("user"));
141         assertEquals("value1", json.get("attr1"));
142         assertEquals("value2", json.get("attr2"));
143         assertEquals("arg1,arg2", StringUtils.join(json.getJSONArray("arguments"), ","));
144         assertEquals("example.Workflow", json.get("className"));
145         assertEquals(exception.toString(), json.get("exception"));
146         assertEquals("interaction1", json.get("interaction"));
147         assertEquals("interactionId1", json.get("interactionId"));
148         assertEquals("returnType1", json.get("returnType"));
149         assertEquals("returnValue1", json.get("returnValue"));
150         assertEquals("subjectId1", json.get("subjectId"));
151         assertEquals("subjectPath1", json.get("subjectPath"));
152         assertEquals("workflowCategory1", json.get("workflowCategory"));
153         assertEquals("workflowName1", json.get("workflowName"));
154         assertEquals(true, json.get("sealed"));
155     }

157     @Test
158     public void testFolderWorkflowEvent() throws Exception {
159         final long timestamp = System.currentTimeMillis();
160 
161         HippoEvent event = new HippoEvent("application1");
162         event.action("action1");
163         event.category("category1");
164         event.message("message1");
165         event.result("result1");
166         event.system(true);
167         event.timestamp(timestamp);
168         event.user("user1");
169         event.set("attr1", "value1");
170         event.set("attr2", "value2");
171         event.sealEvent();
172 
173         FolderWorkflowEvent folderEvent = new FolderWorkflowEvent(event);
174         folderEvent.set("event-type", FolderWorkflowEvent.Type.DELETED);
175         folderEvent.sealEvent();
176 
177         JSONObject json = HippoEventConverter.toJSONObject(folderEvent);
178         assertEquals(FolderWorkflowEvent.class.getName(), json.get("_eventClassName"));
179         assertEquals("application1", json.get("application"));
180         assertEquals("action1", json.get("action"));
181         assertEquals("category1", json.get("category"));
182         assertEquals("message1", json.get("message"));
183         assertEquals("result1", json.get("result"));
184         assertEquals(true, json.get("system"));
185         assertEquals(timestamp, json.get("timestamp"));
186         assertEquals("user1", json.get("user"));
187         assertEquals("value1", json.get("attr1"));
188         assertEquals("value2", json.get("attr2"));
189         assertEquals("DELETED", json.get("type"));
190         assertEquals(true, json.get("sealed"));
191     }
