38  public class HippoEventConsumerTest extends CamelTestSupport {
39  
40      private static final transient Logger LOG = LoggerFactory.getLogger(HippoEventConsumerTest.class);
41  
42      private HippoEventBus eventBus;
43  
44      @Override
45      public void setupResources() throws Exception {
46          HippoServiceRegistry.register(new GuavaHippoEventBus(), HippoEventBus.class);
47          eventBus = HippoServiceRegistry.getService(HippoEventBus.class);
48          assertNotNull(eventBus);
49  
50          super.setupResources();
51      }
52  
53      @Override
54      public void cleanupResources() throws Exception {
55          HippoServiceRegistry.unregister(eventBus, HippoEventBus.class);
56  
57          super.cleanupResources();
58      }
59  
60      @Test
61      public void testHippoEventConsumer() throws Exception {
62          // start consumer thread first
63          HippoEventConsumerThread consumerThread = new HippoEventConsumerThread();
64          consumerThread.start();
65          // wait until the consumer thread has tried to receive event at least once
66          while (consumerThread.getReceiveTrialTimes() < 1) {
67              Thread.sleep(10L);
68          }
69  
70          // now post a HippoEvent
71          HippoEvent event = new HippoEvent("application1");
72          event.action("action1");
73          event.category("category1");
74          event.message("message1");
75          event.result("result1");
76          event.user("user1");
77          eventBus.post(event);
78  
79          // wait until the consumer thread captures an event
80          consumerThread.join();
81  
82          Exchange exchange = consumerThread.getExchange();
83          assertNotNull(exchange);
84  
85          Message message = exchange.getIn();
86          assertNotNull(message);
87          assertTrue(message instanceof HippoEventMessage);
88          JSONObject eventJson = (JSONObject) message.getBody();
89          assertNotNull(eventJson);
90  
91          assertEquals(HippoEvent.class.getName(), eventJson.get("_eventClassName"));
92          assertEquals("action1", eventJson.get("action"));
93          assertEquals("category1", eventJson.get("category"));
94          assertEquals("message1", eventJson.get("message"));
95          assertEquals("result1", eventJson.get("result"));
96          assertEquals("user1", eventJson.get("user"));
97      }
98  
99      @Override
100     protected RouteBuilder createRouteBuilder() throws Exception {
101         return new RouteBuilder() {
102             @Override
103             public void configure() throws Exception {
104                 String uri = "hippoevent:?category=category1&user=user1,user2";
105                 from(uri).to("direct:a");
106             }
107         };
108     }
109 
110     private class HippoEventConsumerThread extends Thread {
111 
112         private Exchange exchange;
113         private int receiveTrialTimes;
114 
115         public HippoEventConsumerThread() {
116             super();
117         }
118 
119         public void run() {
120             while (exchange == null) {
121                 exchange = consumer.receive("direct:a", 10L);
122                 ++receiveTrialTimes;
123 
124                 try {
125                     Thread.sleep(10);
126                 } catch (InterruptedException e) {
127                     break;
128                 }
129             }
130 
131             LOG.debug("HippoEventConsumerThread receive exchange, {} after {} trials", exchange, receiveTrialTimes);
132         }
133 
134         public Exchange getExchange() {
135             return exchange;
136         }
137 
138         public int getReceiveTrialTimes() {
139             return receiveTrialTimes;
140         }
141     }
